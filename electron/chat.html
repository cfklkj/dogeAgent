// 监听 Python 响应 ipcRenderer.on('python-message', (event, message) => { console.log('[接收]', message.type, message); if (message.type === 'chat_response') { const data = message.data; if (data.status === 'success') { addMessage(data.response, false); // 播放 AI 回复 playAIResponse(data.response); } else { addMessage('汪...出错了：' + data.message, false); } } else if (message.type === 'tts_response') { // TTS 响应 const ttsData = message.data; if (ttsData.status === 'success') { console.log('[TTS] 收到音频数据，大小:', ttsData.file_size); // 播放 Base64 音频 if (ttsData.audio_base64) { playBase64Audio(ttsData.audio_base64, ttsData.mime_type); } } else { // TTS 失败，使用浏览器原生 TTS 作为备选 console.error('[TTS] 失败:', ttsData.message); if (ttsData.fallback) { console.log('[TTS] 使用浏览器原生 TTS 作为备选'); browserTTS(ttsData.text); } } } else if (message.type === 'status_response') { const data = message.data; if (data && data.status === 'success' && data.data) { const statusData = data.data; updateAgentStatus( statusData.status, statusData.provider, statusData.model, statusData.message_count, statusData.error_count, statusData.last_update ); } } else if (message.type === 'status_update') { const data = message.data; if (data && data.status === 'success' && data.data) { const statusData = data.data; updateAgentStatus( statusData.status, statusData.provider, statusData.model, statusData.message_count, statusData.error_count, statusData.last_update ); } } });
