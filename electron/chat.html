// 监听 Python 响应 ipcRenderer.on('python-message', (event, message) => { if (message.type === 'chat_response') { const data = message.data; if (data.status === 'success') { addMessage(data.response, false); } else { addMessage('汪...出错了：' + data.message, false); } } else if (message.type === 'status_response' || message.type === 'status_update') { // 处理状态响应/状态推送 const data = message.data; if (data.status === 'success') { const statusData = data.data; updateAgentStatus( statusData.status, statusData.provider, statusData.model, statusData.message_count, statusData.error_count, statusData.last_update ); } } });
